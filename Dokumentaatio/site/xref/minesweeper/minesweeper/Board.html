6   package minesweeper.minesweeper;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  import java.util.Random;
11  
12  /**
13   *
14   * @author Petteri
15   */
16  public class Board {
17      private Tile[][] tiles;
18      
19      public Board(int x, int y, Random random, int mines) {
20          tiles = new Tile[x][y];
21          
22          for (int i = 0; i < x; i++) {
23              for (int j = 0; j < y; j++) {
24                  tiles[i][j] = new Tile();
25              }
26          }
27          setMines(random, mines);
28          countNeighboringMines();
29      }
30      
31      public void setMines(Random random, int mines) {
32          for (int i = 0; i < mines; i++) {
33              int x = random.nextInt(tiles.length);
34              int y = random.nextInt(tiles[0].length);
35  
36              while (tiles[x][y].getIsMine()) {
37                  x = random.nextInt(tiles.length);
38                  y = random.nextInt(tiles[0].length);
39              }
40  
41              tiles[x][y].setIsMine(true);
42          }
43      }
44      
45      public void countNeighboringMines() {
46          for (int i = 0; i < tiles.length; i++) {
47              for (int j = 0; j < tiles[0].length; j++) {
48                  if (!tiles[i][j].getIsMine()) {
49                      int mines = 0;
50                      int xmin = i - 1;
51                      int ymin = j - 1;
52                      int xmax = i + 1;
53                      int ymax = j + 1;
54                      if (i == 0) {
55                          xmin = 0;
56                      }
57                      if (j == 0) {
58                          ymin = 0;
59                      }
60                      if (i == tiles.length - 1) {
61                          xmax = tiles.length - 1;
62                      }
63                      if (j == tiles[0].length - 1) {
64                          ymax = tiles[0].length - 1;
65                      }
66                      
67                      for (int k = xmin; k <= xmax; k++) {
68                          for (int l = ymin; l <= ymax; l++) {
69                              if (tiles[k][l].getIsMine()) {
70                                  mines++;
71                              }
72                          }
73                      }
74                      tiles[i][j].setMines(mines);
75                  } else {
76                      tiles[i][j].setMines(-1);
77                  }
78              }
79          }
80      }
81      
82      public Tile[][] getTiles() {
83          return tiles;
84      }
85  }
