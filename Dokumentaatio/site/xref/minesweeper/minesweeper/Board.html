6   package minesweeper.minesweeper;
7   
8   import java.awt.Color;
9   import java.awt.Dimension;
10  import java.awt.GridLayout;
11  import java.util.ArrayList;
12  import java.util.Random;
13  import javax.swing.JPanel;
14  
15  /**
16   * Board pitää sisällään kaksiulotteisen taulukon kaikista tileistä 
17   * ja suuren osan pelin toiminnallisuudesta.
18   * @author Petteri
19   */
20  public class Board extends JPanel {
21      private Tile[][] tiles;
22      private int flagsLeft;
23      
24      /**
25      * Luokan konstrukstori.
26      * @param x Pelilaudan leveys.
27      * @param y Pelilaudan korkeus.
28      * @param random Käytetään miinojen asettamisessa satunnaisiin sijainteihin.
29      * @param mines Miinojen määrä.
30      */
31      public Board(int x, int y, Random random, int mines) {
32          setLayout(new GridLayout(x, y));
33          setPreferredSize(new Dimension(350, 350));
34          tiles = new Tile[x][y];
35          for (int i = 0; i < x; i++) {
36              for (int j = 0; j < y; j++) {
37                  tiles[i][j] = new Tile(i, j);
38                  this.add(tiles[i][j]);
39              }
40          }
41          setMines(random, mines);
42          countNeighboringMines();
43          flagsLeft = mines;
44      }
45      /**
46      * Asettaa ruuduille miinoja parametreina annetun
47      * mines luvun verran ja randomin mukaisesti.
48      * 
49      * @param random Satunnaislukujuttu
50      * @param mines Miinojen määrä
51      */
52      public void setMines(Random random, int mines) {
53          for (int i = 0; i < mines; i++) {
54              int x = random.nextInt(tiles.length);
55              int y = random.nextInt(tiles[0].length);
56  
57              while (tiles[x][y].getIsMine()) {
58                  x = random.nextInt(tiles.length);
59                  y = random.nextInt(tiles[0].length);
60              }
61  
62              tiles[x][y].setIsMine(true);
63          }
64      }
65      /**
66      * Laskee, kuinka monta miinaa on kunkin ruudun ympärillä
67      * ja asettaa ruudun mines muuttujan.
68      */
69      public void countNeighboringMines() {
70          for (int i = 0; i < tiles.length; i++) {
71              for (int j = 0; j < tiles[0].length; j++) {
72                  if (tiles[i][j].getIsMine()) {
73                      tiles[i][j].setMines(-1);
74                  } else {
75                      int mines = 0;
76                      ArrayList<Tile> neighbors = getNeighbors(tiles[i][j]);
77                      for (int k = 0; k < neighbors.size(); k++) {
78                          if (neighbors.get(k).getIsMine()) {
79                              mines++;
80                          }
81                      }
82                      tiles[i][j].setMines(mines);
83                  }
84              }
85          }
86      }
87      /**
88      * Etsii kaikki vieressä olevat ruudut
89      * syötteenä annetulle ruudulle.
90      *
91      * @param   tile   Ruutu, jonka naapurit halutaan etsiä.
92      * 
93      * @return Lista naapureista.
94      */
95      public ArrayList<Tile> getNeighbors(Tile tile) {
96          ArrayList<Tile> neighbors = new ArrayList<>();
97          int xmin = tile.getTileX() - 1;
98          int ymin = tile.getTileY() - 1;
99          int xmax = tile.getTileX() + 1;
100         int ymax = tile.getTileY() + 1;
101         if (tile.getTileX() == 0) {
102             xmin = 0;
103         }
104         if (tile.getTileY() == 0) {
105             ymin = 0;
106         }
107         if (tile.getTileX() == tiles.length - 1) {
108             xmax = tiles.length - 1;
109         }
110         if (tile.getTileY() == tiles[0].length - 1) {
111             ymax = tiles[0].length - 1;
112         }
113 
114         for (int i = xmin; i <= xmax; i++) {
115             for (int j = ymin; j <= ymax; j++) {
116                 if (i != tile.getTileX() || j != tile.getTileY()) {
117                     neighbors.add(tiles[i][j]);
118                 }
119             }
120         }
121         
122         return neighbors;
123     }
124     /**
125     * Paljastaa ruudun ja kutsuu itsensä paljastamaan ruudun naapurit, 
126     * jos ruudun vieressä ei ole yhtään miinaa.
127     *
128     * @param   tile   Ruutu, joka paljastetaan.
129     */
130     public void revealTile(Tile tile) {
131         if (tile.getIsHidden() && !tile.getIsFlagged()) {
132             if (tile.getIsMine()) {
133                 lose();
134             } else {
135                 tile.reveal();
136                 if (tile.getMines() == 0) {
137                     ArrayList<Tile> neighbors = getNeighbors(tile);
138                     for (int i = 0; i < neighbors.size(); i++) {
139                         revealTile(neighbors.get(i));
140                     }
141                 }
142             }
143         }
144     }
145     /**
146     * Paljastaa kaikki ruudut ja vaihtaa miinan värin parametrin mukaisesti. 
147     * Käytetään pelin päättyessä.
148     * 
149     * @param color Haluttu väri. lightGray on "oletusväri", RED hävitessä, GREEN voittaessa.
150     */
151     public void revealAllTiles(Color color) {
152         for (int i = 0; i < tiles.length; i++) {
153             for (int j = 0; j < tiles[0].length; j++) {
154                 tiles[i][j].setIsHidden(false);
155                 if (tiles[i][j].getIsMine()) {
156                     tiles[i][j].setBackground(color);
157                 }
158             }
159         }
160     }
161     /**
162     * Laittaa tileen lipun tai ottaa sen pois, jos sillä on sellainen.
163     * Sitten päivittää lippujen määrää suoritetun toiminnon mukaisesti.
164     * @param tile Tile, jolle toiminto suoritetaan.
165     */
166     public void toggleFlag(Tile tile) {
167         if (tile.putFlag()) {
168             flagsLeft--;
169         } else {
170             flagsLeft++;
171         }
172         if (flagsLeft == 0) {
173             checkWin();
174         }
175     }
176     /**
177     * Tarkistaa, onko olemassa ruutuja, joissa on miina, muttei lippua.
178     * Jos on, paljastetaan kaikki ruudut ja väritetään ne vihreiksi.
179     * Muussa tapauksessa ei tehdä mitään.
180     */
181     public void checkWin() {
182         boolean winning = true;
183         for (int i = 0; i < tiles.length; i++) {
184             for (int j = 0; j < tiles[0].length; j++) {
185                 if (!tiles[i][j].getIsFlagged() && tiles[i][j].getIsMine()) {
186                     winning = false;
187                     i = tiles.length;
188                     break;
189                 }
190             }
191         }
192         if (winning) {
193             revealAllTiles(Color.GREEN);
194         }
195     }
196     /**
197     * Kutsuu revealAllTiles() metodin parametrilla Color.RED.
198     * Käytetään, kun pelaaja häviää pelin.
199     */
200     public void lose() {
201         revealAllTiles(Color.RED);
202     }
203     
204     public Tile[][] getTiles() {
205         return tiles;
206     }
207     
208     public int getFlagsLeft() {
209         return flagsLeft;
210     }
211     
212     public void setFlagsLeft(int i) {
213         flagsLeft = i;
214     }
215     /**
216     * Palauttaa tietyssä sijainnissa olevan Tilen.
217     * @param x Tilen X -koordinaatti.
218     * @param y Tilen Y -koordinaatti.
219     * @return Palauttaa Tilen, joka löydettiin annetuilla koordinaateilla.
220     */
221     public Tile getTileAt(int x, int y) {
222         return tiles[x][y];
223     }
